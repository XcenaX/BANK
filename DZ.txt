create Function Func_1()returns int as
begin
return (
select count(B.b_id) from dbo.CategoryTable C, dbo.BookTable B
where C.cat_id = null
)
end



create Function Func_2() returns int as
begin
return (
select count(B.b_id) from dbo.BookTable B, dbo.CategoryTable Cat, dbo.SubcatTable Sub
where B.category_id = Cat.cat_id and B.subcat_id = Sub.subcat_id
)
end


create Function Func_3(@name nvarchar, @theme nvarchar, @category nvarchar, @edition nvarchar) 
returns nvarchar as
begin
return (
select B.b_nm from dbo.BookTable B
where B.b_nm like @name
and B.b_edition like @edition
and B.b_subcat like @category
and B.b_desc like @theme
)
end



create Function Min_Num(@num1 int, @num2 int , @num3 int) returns int as
begin
	declare @MinNum int
	if @num1 > @num2 
begin
	if @num1 > @num3 begin select @MinNum = @num1 end
end
	else if @num2 > @num3 begin select @MinNum = @num2 end
	else  begin select @MinNum = @num3 end
	return @MinNum
end
